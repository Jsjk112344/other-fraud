    3. Create `backend/api/investigate.py`:
       ```python
       from fastapi import APIRouter, Request
       from sse_starlette.sse import EventSourceResponse
       from models.events import InvestigateRequest
       from mock.pipeline import run_mock_investigation
       from urllib.parse import urlparse
       import json

       router = APIRouter()

       def validate_url(url: str) -> bool:
           try:
               result = urlparse(url)
               return all([result.scheme in ('http', 'https'), result.netloc])
           except Exception:
               return False

       @router.post("/api/investigate")
       async def investigate(request: Request, body: InvestigateRequest):
           if not validate_url(body.url):
               async def error_generator():
                   yield {
                       "event": "error",
                       "data": json.dumps({"error": "Could not detect listing. Please paste a valid marketplace URL."})
                   }
               return EventSourceResponse(error_generator())

           async def event_generator():
               async for event in run_mock_investigation(body.url):
                   if await request.is_disconnected():
                       break
                   yield {
                       "event": event.step,
                       "data": json.dumps(event.model_dump())
                   }

           return EventSourceResponse(event_generator())
       ```
